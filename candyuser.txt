const Sequelize = require('sequelize');
function candyuser(sequelize)
{
	return sequelize.define('candyuser',
	{
		ethAddr:{
			type:Sequelize.STRING(42),
			allowNull:false,
			primaryKey:true
		},
		privateCode:{
			type:Sequelize.STRING(32),
			allowNull:false,
		},
		joinTelegram:{
			type:Sequelize.BOOLEAN,
			allowNull:false,
			defaultValue:false
		},
		telegramId:{
			type:Sequelize.STRING,
			allowNull:false,
			defaultValue:''
		},
		twitterTaskSign:{
			type:Sequelize.BOOLEAN,
			allowNull:false,
			defaultValue:true
		},
		twitterShareTimes:{
			type:Sequelize.INTEGER,
			allowNull:false,
			defaultValue:0
		},
		facebookTaskSign:{
			type:Sequelize.BOOLEAN,
			allowNull:false,
			defaultValue:true
		},
		facebookShareTimes:{
			type:Sequelize.INTEGER,
			allowNull:false,
			defaultValue:0
		},
		mailVerify:{
			type:Sequelize.BOOLEAN,
			allowNull:false,
			defaultValue:false
		},
		mail:{
			type:Sequelize.STRING,
			allowNull:false,
			defaultValue:''
		},
		verifyAll:{
			type:Sequelize.BOOLEAN,
			allowNull:false,
			defaultValue:false
		},
		inviteCount:{
			type:Sequelize.INTEGER,
			allowNull:false,
			defaultValue:0
		},
		inviterCode:{
			type:Sequelize.STRING(32),
			allowNull:false,
			defaultValue:''
		},
		totalCoin:{
			type:Sequelize.DECIMAL,
			allowNull:false,
			defaultValue:0
		},
		payCoin:{
			type:Sequelize.DECIMAL,
			allowNull:false,
			defaultValue:0
		},
		updated:{
			type:Sequelize.DATE(6),
			allowNull:false,
			defaultValue:Sequelize.NOW
		}
	},
	{
		timestamps:false,
		freezeTableName:true
	})
}

module.exports = candyuser;
